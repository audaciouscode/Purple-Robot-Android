Disables automatic fires for trigger identified by triggerId.

Arguments

identifier - Unique identifier of the trigger to disable.

Return Values

None

JavaScript Example

var dateTrigger = {
    "identifier": "hourly-chime",
    "name": "Hourly Chime",
    "type": "datetime",
    "action": "PurpleRobot.playDefaultTone();",
    "datetime_start": "20150327T130000",
    "datetime_end": "20150327T130100",
    "datetime_random": false,
    "datetime_repeat": "FREQ=HOURLY;INTERVAL=1;",
    "fire_on_boot": true
};

PurpleRobot.updateTrigger('hourly-chime', dateTrigger);

PurpleRobot.disableTrigger("hourly-chime");

Scheme Example

(pr-update-trigger '(("identifier" . "hourly-chime")
                     ("name" . "Hourly Chime")
                     ("type" . "datetime")
                     ("action" . "(pr-play-default-tone)")
                     ("datetime_start" . "20150327T130000")
                     ("datetime_end" . "20150327T130100")
                     ("datetime_random" . #f)
                     ("datetime_repeat" . "FREQ=HOURLY;INTERVAL=1;")
                     ("fire_on_boot" . #t)))

(pr-disable-trigger "hourly-chime")
